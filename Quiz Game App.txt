<script>
  const quizData = [
    {
      type: 'single',
      question: 'Which planet is known as the Red Planet?',
      options: ['Earth', 'Mars', 'Jupiter', 'Venus'],
      answer: 'Mars'
    },
    {
      type: 'multiple',
      question: 'Select all programming languages:',
      options: ['Python', 'HTML', 'JavaScript', 'CSS'],
      answer: ['Python', 'JavaScript']
    },
    {
      type: 'fill',
      question: 'What is the capital of France?',
      answer: 'Paris'
    },
    {
      type: 'single',
      question: 'Which of these is NOT a primary color?',
      options: ['Red', 'Blue', 'Green', 'Yellow'],
      answer: 'Green'
    },
    {
      type: 'fill',
      question: 'What does ‚ÄúCPU‚Äù stand for?',
      answer: 'Central Processing Unit'
    }
  ];

  const questionContainer = document.getElementById('question-container');
  const nextBtn = document.getElementById('next-btn');
  const resultDiv = document.getElementById('result');

  let currentQuestion = 0;
  let score = 0;

  function loadQuestion() {
    const q = quizData[currentQuestion];
    questionContainer.innerHTML = `
      <div class="question">${q.question}</div>
    `;

    if (q.type === 'single') {
      const optionsHTML = q.options.map(
        opt => `<label><input type="radio" name="option" value="${opt}"> ${opt}</label>`
      ).join('');
      questionContainer.innerHTML += `<div class="options">${optionsHTML}</div>`;
    } else if (q.type === 'multiple') {
      const optionsHTML = q.options.map(
        opt => `<label><input type="checkbox" name="option" value="${opt}"> ${opt}</label>`
      ).join('');
      questionContainer.innerHTML += `<div class="options">${optionsHTML}</div>`;
    } else if (q.type === 'fill') {
      questionContainer.innerHTML += `<input type="text" id="fillAnswer" placeholder="Type your answer here..." />`;
    }

    nextBtn.textContent = currentQuestion === quizData.length - 1 ? 'Finish' : 'Next';
  }

  function checkAnswer() {
    const q = quizData[currentQuestion];
    let userAnswer;

    if (q.type === 'single') {
      const selected = document.querySelector('input[name="option"]:checked');
      if (!selected) return alert('Please select an answer!');
      userAnswer = selected.value;
      if (userAnswer === q.answer) score++;
    }

    else if (q.type === 'multiple') {
      const selected = Array.from(document.querySelectorAll('input[name="option"]:checked')).map(el => el.value);
      if (selected.length === 0) return alert('Please select at least one answer!');
      if (JSON.stringify(selected.sort()) === JSON.stringify(q.answer.sort())) score++;
    }

    else if (q.type === 'fill') {
      const userInput = document.getElementById('fillAnswer').value.trim();
      if (!userInput) return alert('Please enter an answer!');
      if (userInput.toLowerCase() === q.answer.toLowerCase()) score++;
    }

    currentQuestion++;
    if (currentQuestion < quizData.length) {
      loadQuestion();
    } else {
      showResult();
    }
  }

  function showResult() {
    document.getElementById('quiz').classList.add('hidden');
    resultDiv.classList.remove('hidden');
    resultDiv.innerHTML = `
      üèÅ Quiz Complete!<br>
      Your Score: <strong>${score}</strong> / ${quizData.length}
    `;
  }

  nextBtn.addEventListener('click', checkAnswer);

  loadQuestion();
</script>
